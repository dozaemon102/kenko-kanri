## Pi デプロイ（人間）

```bash
# MySQL
DROP DATABASE IF EXISTS sanpo_ban;
CREATE DATABASE kenko_kanri CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
DROP USER IF EXISTS 'sanpo'@'localhost';
CREATE USER 'kenko'@'localhost' IDENTIFIED BY 'kenko';
GRANT ALL ON kenko_kanri.* TO 'kenko'@'localhost';
FLUSH PRIVILEGES;

# アプリ
cd ~/kenko-kanri/src/backend
# .env の DATABASE_URL を mysql+pymysql://kenko:kenko@localhost:3306/kenko_kanri に更新
uv run alembic upgrade head
cd ../frontend && npm run build
sudo systemctl restart kenko-kanri
```
